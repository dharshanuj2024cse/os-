1.LRU Page Replacement with Frame Table
#include <stdio.h>

int main() {

    int n,f,fault=0,hit=0,count=0;

    printf("Enter number of pages: ");
    scanf("%d",&n);

    int p[n];

    printf("Enter pages: ");
    for(int i=0;i<n;i++)
        scanf("%d",&p[i]);

    printf("Enter number of frames: ");
    scanf("%d",&f);

    int frame[f],time[f];

    for(int i=0;i<f;i++) {
        frame[i]=-1;
        time[i]=0;
    }

    printf("\nPage\tFrames\n");

    for(int i=0;i<n;i++) {

        int found=0;

        for(int j=0;j<f;j++) {

            if(frame[j]==p[i]) {

                hit++;
                count++;
                time[j]=count;
                found=1;
            }
        }

        if(!found) {

            int lru=0;

            for(int j=1;j<f;j++) {

                if(time[j]<time[lru])
                    lru=j;
            }

            count++;

            frame[lru]=p[i];
            time[lru]=count;

            fault++;
        }

        // Display frame table
        printf("%d\t",p[i]);

        for(int j=0;j<f;j++) {

            if(frame[j]==-1)
                printf("- ");
            else
                printf("%d ",frame[j]);
        }

        printf("\n");
    }

    printf("\nPage Faults = %d\n",fault);
    printf("Page Hits = %d\n",hit);

    return 0;
}

    int frame[f];

    for(int i=0;i<f;i++)
        frame[i]=-1;

    printf("\nPage\t");

    for(int i=0;i<f;i++)
        printf("F%d\t",i+1);

    printf("\n");

    for(int i=0;i<n;i++) {

        int found=0;

        for(int j=0;j<f;j++) {

            if(frame[j]==p[i]) {
                found=1;
                hit++;
            }
        }

        if(!found) {

            frame[k]=p[i];

            k=(k+1)%f;

            fault++;
        }

        printf("%d\t",p[i]);

        for(int j=0;j<f;j++) {

            if(frame[j]==-1)
                printf("-\t");
            else
                printf("%d\t",frame[j]);
        }

        printf("\n");
    }

    printf("\nPage Faults = %d\n",fault);
    printf("Page Hits = %d\n",hit);

    return 0;
} 


3.Bankers algorithm:
#include <stdio.h>

int main() {

    int p,r,i,j,count=0;

    printf("Enter processes and resources: ");
    scanf("%d%d",&p,&r);

    int alloc[p][r],max[p][r],need[p][r];
    int avail[r],finish[p],safe[p];

    printf("Enter Allocation Matrix:\n");
    for(i=0;i<p;i++)
        for(j=0;j<r;j++)
            scanf("%d",&alloc[i][j]);

    printf("Enter Max Matrix:\n");
    for(i=0;i<p;i++)
        for(j=0;j<r;j++)
            scanf("%d",&max[i][j]);

    printf("Enter Available Resources:\n");
    for(i=0;i<r;i++)
        scanf("%d",&avail[i]);

    for(i=0;i<p;i++) {

        finish[i]=0;

        for(j=0;j<r;j++)
            need[i][j]=max[i][j]-alloc[i][j];
    }

    while(count<p) {

        int found=0;

        for(i=0;i<p;i++) {

            if(finish[i]==0) {

                int possible=1;

                for(j=0;j<r;j++) {

                    if(need[i][j]>avail[j])
                        possible=0;
                }

                if(possible) {

                    for(j=0;j<r;j++)
                        avail[j]+=alloc[i][j];

                    safe[count++]=i;

                    finish[i]=1;

                    found=1;
                }
            }
        }

        if(!found)
            break;
    }

    if(count==p) {

        printf("Safe Sequence: ");

        for(i=0;i<p;i++)
            printf("P%d ",safe[i]);
    }

    else
        printf("Deadlock");

    return 0;
}

4.BEST FIT:
#include<stdio.h>

int main() {
    int nb,np,b[10],p[10];

    printf("Enter blocks and processes: ");
    scanf("%d%d",&nb,&np);

    printf("Enter block sizes:\n");
    for(int i=0;i<nb;i++)
        scanf("%d",&b[i]);

    printf("Enter process sizes:\n");
    for(int i=0;i<np;i++)
        scanf("%d",&p[i]);

    for(int i=0;i<np;i++) {
        int best=-1;

        for(int j=0;j<nb;j++) {
            if(b[j]>=p[i] && (best==-1 || b[j]<b[best]))
                best=j;
        }

        if(best!=-1) {
            printf("P%d -> B%d\n",i+1,best+1);
            b[best]-=p[i];
        }
        else
            printf("P%d Not Allocated\n",i+1);
    }

    return 0;
}

6.FCFS:
#include <stdio.h>

int main() {

    int n;

    printf("Enter number of processes: ");
    scanf("%d",&n);

    int AT[n],BT[n],CT[n],TAT[n],WT[n];

    for(int i=0;i<n;i++) {

        printf("Enter AT and BT for P%d: ",i+1);
        scanf("%d%d",&AT[i],&BT[i]);
    }

    CT[0]=AT[0]+BT[0];

    for(int i=1;i<n;i++) {

        if(CT[i-1] < AT[i])
            CT[i]=AT[i]+BT[i];
        else
            CT[i]=CT[i-1]+BT[i];
    }

    float avgTAT=0,avgWT=0;

    for(int i=0;i<n;i++) {

        TAT[i]=CT[i]-AT[i];
        WT[i]=TAT[i]-BT[i];

        avgTAT += TAT[i];
        avgWT += WT[i];
    }

    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");

    for(int i=0;i<n;i++) {

        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
               i+1,AT[i],BT[i],CT[i],TAT[i],WT[i]);
    }

    printf("\nAverage TAT = %.2f\n",avgTAT/n);
    printf("Average WT = %.2f\n",avgWT/n);

    return 0;
}

7.SJF:
#include <stdio.h>

int main() {

    int n,time=0,completed=0;

    printf("Enter number of processes: ");
    scanf("%d",&n);

    int AT[n],BT[n],CT[n],TAT[n],WT[n],done[n];

    for(int i=0;i<n;i++) {

        printf("Enter AT and BT for P%d: ",i+1);
        scanf("%d%d",&AT[i],&BT[i]);

        done[i]=0;
    }

    while(completed<n) {

        int min=-1;

        for(int i=0;i<n;i++) {

            if(AT[i]<=time && done[i]==0) {

                if(min==-1 || BT[i]<BT[min])
                    min=i;
            }
        }

        if(min==-1)
            time++;

        else {

            time += BT[min];

            CT[min]=time;

            done[min]=1;

            completed++;
        }
    }

    float avgTAT=0,avgWT=0;

    for(int i=0;i<n;i++) {

        TAT[i]=CT[i]-AT[i];
        WT[i]=TAT[i]-BT[i];

        avgTAT += TAT[i];
        avgWT += WT[i];
    }

    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");

    for(int i=0;i<n;i++) {

        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
               i+1,AT[i],BT[i],CT[i],TAT[i],WT[i]);
    }

    printf("\nAverage TAT = %.2f\n",avgTAT/n);
    printf("Average WT = %.2f\n",avgWT/n);

    return 0;
}

9.RR:
#include <stdio.h>

int main() {

    int n,tq,time=0,completed=0;

    printf("Enter number of processes: ");
    scanf("%d",&n);

    printf("Enter Time Quantum: ");
    scanf("%d",&tq);

    int AT[n],BT[n],RT[n],CT[n],TAT[n],WT[n];

    for(int i=0;i<n;i++) {

        printf("Enter AT and BT for P%d: ",i+1);
        scanf("%d%d",&AT[i],&BT[i]);

        RT[i]=BT[i];
    }

    while(completed<n) {

        int done=1;

        for(int i=0;i<n;i++) {

            if(RT[i]>0 && AT[i]<=time) {

                done=0;

                if(RT[i]>tq) {

                    time += tq;
                    RT[i] -= tq;
                }

                else {

                    time += RT[i];

                    RT[i]=0;

                    CT[i]=time;

                    completed++;
                }
            }
        }

        if(done)
            time++;
    }

    float avgTAT=0,avgWT=0;

    for(int i=0;i<n;i++) {

        TAT[i]=CT[i]-AT[i];
        WT[i]=TAT[i]-BT[i];

        avgTAT += TAT[i];
        avgWT += WT[i];
    }

    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");

    for(int i=0;i<n;i++) {

        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
               i+1,AT[i],BT[i],CT[i],TAT[i],WT[i]);
    }

    printf("\nAverage TAT = %.2f\n",avgTAT/n);
    printf("Average WT = %.2f\n",avgWT/n);

    return 0;
}
